Player

		int mCredits                // Sätts till ATSTART_CREDITS i konstruktorn

	metoder
		bool buyAnimal( Animal pAnimal )  // Köper från Store, tas från affärens lista, läggs i spelarens lista

		void printLivestock()  // Skriver ut vilka djur spelaren äger
		void printCredits()    // Skriver ut hur mycket pengar har spelaren
